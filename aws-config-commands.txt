# AWS Configuration Commands for Firebase Functions
# Replace the placeholder values with your actual AWS credentials

# Set AWS Region (e.g., us-east-1, eu-west-1, ap-southeast-1, etc.)
firebase functions:config:set aws.region="YOUR_AWS_REGION"

# Set AWS Access Key ID
firebase functions:config:set aws.access_key_id="YOUR_***REMOVED***"

# Set AWS Secret Access Key
firebase functions:config:set aws.secret_access_key="YOUR_***REMOVED***"

# Set S3 Bucket Name
firebase functions:config:set aws.bucket_name="YOUR_S3_BUCKET_NAME"

# After setting all configurations, verify with:
firebase functions:config:get

# Then deploy the functions:
firebase deploy --only functions
